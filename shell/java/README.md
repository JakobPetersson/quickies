# Switch java versions

```bash
alias jdk8='export JAVA_HOME=$(/usr/libexec/java_home -v 1.8)'
alias jdk9='export JAVA_HOME=$(/usr/libexec/java_home -v 9)'
alias jdk10='export JAVA_HOME=$(/usr/libexec/java_home -v 10)'
alias jdk11='export JAVA_HOME=$(/usr/libexec/java_home -v 11)'
alias jdk14='export JAVA_HOME=$(/usr/libexec/java_home -v 14)'
```
